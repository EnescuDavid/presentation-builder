API-Authentifizierung


import requests
from datetime import datetime

class APIClient:
    """Basis-Client für die REST-API-Authentifizierung."""

    def __init__(self, base_url, api_key):
        self.base_url = base_url
        self.session = requests.Session()
        self.session.headers.update({
            'Authorization': f'Bearer {api_key}',
            'Content-Type': 'application/json'
        })

    def get(self, endpoint):
        """GET-Anfrage mit automatischer Fehlerbehandlung."""
        response = self.session.get(f'{self.base_url}{endpoint}')
        response.raise_for_status()
        return response.json()

Datenverarbeitungs-Pipeline


# Schritt 1: Daten laden und validieren
import pandas as pd
df = pd.read_csv('umsatzdaten_2025.csv')
df = df.dropna(subset=['betrag', 'datum'])

# Schritt 2: Transformation und Aggregation
df['quartal'] = pd.to_datetime(df['datum']).dt.quarter
df['jahr'] = pd.to_datetime(df['datum']).dt.year
zusammenfassung = df.groupby(['jahr', 'quartal']).agg(
    gesamt_umsatz=('betrag', 'sum')
)

# Schritt 3: Ergebnis exportieren
zusammenfassung.to_excel(
    'quartalsbericht.xlsx',
    sheet_name='Übersicht',
    index=True,
    float_format='%.2f'
)
print(f'Bericht erstellt: {len(zusammenfassung)} Quartale')
