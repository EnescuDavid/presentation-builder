import requests
from datetime import datetime

def hole_umsatzdaten(quartal: str, jahr: int) -> dict:
    """Laed Umsatzdaten fuer ein Quartal aus der ERP-API."""
    basis_url = "https://api.unternehmen.de/v2"
    endpunkt = f"{basis_url}/finanzen/umsatz"

    antwort = requests.get(endpunkt, params={
        "quartal": quartal,
        "jahr": jahr,
        "waehrung": "EUR"

    antwort.raise_for_status()
