**5. Add deep merge utility and lazy chart init helper** (per D-03/D-24):
```javascript
// --- Deep Merge Utility ---
function deepMerge(target, source) {
  var result = Object.assign({}, target);
  for (var key in source) {
    if (source[key] && typeof source[key] === 'object' && !Array.isArray(source[key])) {
      result[key] = deepMerge(result[key] || {}, source[key]);
    } else {
      result[key] = source[key];
    }
  }
  return result;
}

// --- Lazy Chart Init (per D-03/D-24) ---
var chartInstances = {};

function initSlideCharts(slide) {
  if (typeof Chart === 'undefined') return;
  var canvases = slide.querySelectorAll('canvas[data-chart-config]');
  canvases.forEach(function(canvas) {
    var chartId = canvas.id;
    if (!chartId || chartInstances[chartId]) return; // Animate-once per D-02
    var configStr = canvas.getAttribute('data-chart-config');
    var config = JSON.parse(configStr);
    var defaults = getConsultingDefaults();
    config.options = deepMerge(defaults, config.options || {});
    // Auto-assign token colors if not explicitly set
    var colors = getChartColors();
    config.data.datasets.forEach(function(ds, i) {
      if (!ds.backgroundColor) ds.backgroundColor = colors[i % colors.length];
      if (!ds.borderColor) ds.borderColor = colors[i % colors.length];
    });
    chartInstances[chartId] = new Chart(canvas, config);
  });
}
```
