    // Fragment reveal handler for progressive chart datasets (per D-06)
    // When a fragment is shown, unhide the corresponding dataset and update.
    Reveal.on('fragmentshown', function(event) {
      var slide = event.fragment.closest('section');
      var canvas = slide ? slide.querySelector('canvas[data-chart-config]') : null;
      if (!canvas || !canvas.id) return;
      var chart = chartInstances[canvas.id];
      if (!chart) return;
      var idx = parseInt(event.fragment.getAttribute('data-fragment-index'), 10);
      if (chart.data.datasets[idx]) {
        chart.data.datasets[idx].hidden = false;
        chart.update('active');
      }
    });
  </script>

-->
