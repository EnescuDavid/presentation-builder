```css
  padding-bottom: var(--spacing-xs);
  border-bottom: var(--theme-heading-border);
  display: inline-block;
  margin-bottom: var(--spacing-lg);
}
```

```css
.reveal .slides section ul { list-style: none; padding-left: var(--spacing-lg); }

}
```

```css
}
```
