**2. Create tools/extract-theme.js** with this structure:

```
Usage: node tools/extract-theme.js <path-to-pptx> [output-dir]
Default output-dir: ./extracted
```

The script must:

a) **Parse CLI args**: pptxPath (required), outputDir (default: ./extracted). Print usage and exit 1 if no pptxPath.

b) **Open PPTX as ZIP** using `new AdmZip(pptxPath)`.

c) **Find theme XML**: First try `ppt/theme/theme1.xml`. If not found, scan ZIP entries for any file matching `ppt/theme/theme*.xml` and use the first. If no theme found, print error and exit 1.

d) **Parse theme XML** using `new XMLParser({ ignoreAttributes: false, attributeNamePrefix: '@_' })`.

e) **Extract 12 color slots** from `a:theme > a:themeElements > a:clrScheme`. For each of the 12 slots (dk1, lt1, dk2, lt2, accent1, accent2, accent3, accent4, accent5, accent6, hlink, folHlink):
   - Check for `a:srgbClr` element, read `@_val` attribute, prepend `#`
   - If not found, check for `a:sysClr` element, read `@_lastClr` attribute, prepend `#`
   - If neither found, set to null and log warning

f) **Extract 2 font families** from `a:fontScheme`:
   - major: `a:majorFont > a:latin @_typeface`
   - minor: `a:minorFont > a:latin @_typeface`

g) **Include font fallback lookup table** (the FONT_FALLBACKS object from the research, covering 20+ common corporate fonts mapped to Google Fonts equivalents). Add a `googleFontFallback` field to each font in the output.

h) **Extract images** from `ppt/media/` entries in the ZIP:
   - Create output directory with `fs.mkdirSync(outputDir, { recursive: true })`
   - For each ZIP entry starting with `ppt/media/`:
     - Extract file to outputDir
     - Record name, originalPath, size in bytes

i) **Count slide masters and extract layout names**:
   - Count files matching `ppt/slideMasters/slideMaster*.xml`
   - Parse `ppt/slideLayouts/` entries to get layout names if available (use the relationship XML if present, otherwise just count)

j) **Generate recommended color mapping** to framework tokens:
   Use the mapping from the research (Pattern 4):
   ```
   dk1 -> --color-text
   lt1 -> --color-background
   dk2 -> --color-primary
   lt2 -> --color-surface
   accent1 -> --color-accent, --color-chart-1
   accent2 -> --color-danger, --color-chart-4
   accent3 -> --color-success, --color-chart-2
   accent4 -> --color-chart-3
   accent5 -> --color-secondary
   accent6 -> --color-chart-5
   ```
   Also compute derived tokens:
   - --color-text-muted: dk2 value with opacity note
   - --color-border: lt2 value with "darken slightly" note
   - --color-highlight: accent1 value with "lighten to 10% opacity" note
   - --color-callout: accent2 value with "lighten to 10% opacity" note

k) **Write JSON output** to `{outputDir}/extracted-theme.json` with structure:
   ```json
   {
     "source": "filename.pptx",
     "extracted": "ISO 8601 timestamp",
     "colors": { ...12 slots },
     "fonts": {
       "major": { "latin": "FontName", "googleFallback": "GoogleFont" },
       "minor": { "latin": "FontName", "googleFallback": "GoogleFont" }
     },
     "images": [...],
     "slideMasters": { "count": N, "layouts": [...] },
     "colorMapping": { ...recommended token mappings with hex values }
   }
   ```

l) **Print JSON to stdout** as well (so Claude can read it directly).

m) **Error handling**: Wrap in try/catch. Print clear error messages for: file not found, not a valid ZIP, no theme XML found, parse failures. Exit 1 on error.

**3. Create tools/test-fixtures/verify-extraction.js:**

```javascript
// Verifies extract-theme.js output against known test fixture values
const fs = require('fs');
const path = require('path');
const { execSync } = require('child_process');

const EXPECTED = {
  colors: {
    dk1: '#1A1A1A',   // From sysClr lastClr
    lt1: '#FAFAFA',
    dk2: '#2B4C7E',
    lt2: '#E8ECF1',
    accent1: '#3498DB',
    accent2: '#E74C3C',
    accent3: '#2ECC71',
    accent4: '#9B59B6',
    accent5: '#1ABC9C',
    accent6: '#F39C12',
    hlink: '#2980B9',
    folHlink: '#8E44AD'
  },
  fonts: {
    major: 'Calibri Light',
    minor: 'Calibri'
  }
};

// Run extraction
const pptxPath = path.join(__dirname, 'sample-theme.pptx');
const outputDir = path.join(__dirname, 'test-output');

try {
  execSync(`node ${path.join(__dirname, '..', 'extract-theme.js')} "${pptxPath}" "${outputDir}"`, { stdio: 'pipe' });
} catch (e) {
  console.error('FAIL: Extraction script errored:', e.stderr?.toString());
  process.exit(1);
}

// Read output
const jsonPath = path.join(outputDir, 'extracted-theme.json');
if (!fs.existsSync(jsonPath)) {
  console.error('FAIL: extracted-theme.json not created');
  process.exit(1);
}

const result = JSON.parse(fs.readFileSync(jsonPath, 'utf8'));
let failures = 0;

// Verify colors
for (const [slot, expected] of Object.entries(EXPECTED.colors)) {
  const actual = result.colors[slot];
  if (actual?.toUpperCase() !== expected.toUpperCase()) {
    console.error(`FAIL: colors.${slot} expected ${expected}, got ${actual}`);
    failures++;
  }
}

// Verify fonts
if (result.fonts.major?.latin !== EXPECTED.fonts.major) {
  console.error(`FAIL: fonts.major expected ${EXPECTED.fonts.major}, got ${result.fonts.major?.latin}`);
  failures++;
}
if (result.fonts.minor?.latin !== EXPECTED.fonts.minor) {
  console.error(`FAIL: fonts.minor expected ${EXPECTED.fonts.minor}, got ${result.fonts.minor?.latin}`);
  failures++;
}

// Verify images extracted
const logoExists = fs.existsSync(path.join(outputDir, 'test-logo.png'));
if (!logoExists) {
  console.error('FAIL: test-logo.png not extracted to output dir');
  failures++;
}

// Verify colorMapping present
if (!result.colorMapping || typeof result.colorMapping !== 'object') {
  console.error('FAIL: colorMapping not present in output');
  failures++;
}

// Verify source field
if (!result.source) {
  console.error('FAIL: source field missing');
  failures++;
}

// Cleanup
fs.rmSync(outputDir, { recursive: true, force: true });

if (failures === 0) {
  console.log('PASS: All extraction checks passed');
  process.exit(0);
} else {
  console.error(`FAIL: ${failures} check(s) failed`);
  process.exit(1);
}
```
